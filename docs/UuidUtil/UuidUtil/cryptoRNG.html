Crypto-Strong RNG. All platforms, unknown speed, cryptographically strong (theoretically)

static List<int> cryptoRNG() {
  var b = new List<int>(16);
  var rand = Random.secure();
  for (var i = 0; i < 16; i++) {
    b[i] = rand.nextInt(1 << 8);
  }
  return b;
}
